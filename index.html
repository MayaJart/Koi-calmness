let waterTime = 0; function drawWater() { // base water color const grad = ctx.createLinearGradient(0, 0, 0, canvas.height); grad.addColorStop(0, '#1bb6d8'); grad.addColorStop(1, '#0b4f6c'); ctx.fillStyle = grad; ctx.fillRect(0, 0, canvas.width, canvas.height); waterTime += 0.01; ctx.save(); ctx.globalAlpha = 0.25; ctx.globalCompositeOperation = 'screen'; ctx.strokeStyle = 'rgba(255,255,255,0.6)'; ctx.lineWidth = 1.5; const spacing = 40; for (let y = 0; y < canvas.height + spacing; y += spacing) { ctx.beginPath(); for (let x = 0; x <= canvas.width; x += 20) { const wave = Math.sin(x * 0.02 + waterTime * 3) * 6 + Math.sin(y * 0.05 + waterTime * 2) * 4; ctx.lineTo(x, y + wave); } ctx.stroke(); } ctx.restore(); }
